Alias for the apt manager:
	alias sync="sudo apt update"
	alias upgrade="sudo apt upgrade -y"
	alias dist_upgrade="sudo apt dist-upgrade"
	alias clean="sudo apt autoremove && sudo apt autoclean"
	alias install="sudo apt intsall -y "
	alias search="apt search "


Alias for the PacMan manager:
        alias sync="sudo pacman -Syy"
        alias upgrade="sudo pacman -Syyu"
        alias search_repo="pacman -Ss "
        alias search_system="pacman -Qi "
        alias list_system="pacman -Ql | less"
        alias list_orphans="pacman -Qdt"
        alias remove_orphans="sudo pacman -Rsn "
        alias remove_package="sudo pacman -Rs "
